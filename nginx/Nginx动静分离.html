编辑 Nginx 配置文件

# 删除反向代理配置文件
[root@localhost ~]# rm /tmp/nginx/n80/conf.d/proxy.conf

# 更改默认配置文件
[root@localhost ~]# vim /tmp/nginx/n80/conf.d/default.conf

server {
    listen       80;
    server_name  ahri.com;

    location / {
        root   /usr/share/nginx/html;
        index  index.html index.htm;
    }

    location /images/ {
        root   /usr/share/nginx/html;
        autoindex       on;
    }
}
